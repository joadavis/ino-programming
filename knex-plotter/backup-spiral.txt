int SERVO_X1_HOME = 5;

void setup() 
{ 
  //myservo.attach(9);  // attaches the servo on pin 9 to the servo object 
  servo_pen.attach(11);
  servo_y.attach(10);
  servo_x1.attach(6);
  servo_x2.attach(5);
  
  Serial.println("Setting up, pen=11, y=10, x= 6 and 5");
  
  hard_zero();
} 
 
 
void loop() 
{ 
  action = 4;
  if (action == 0) 
  {
    test_box();
  }
  else if (action == 1)
  {
    sweep();
  }
  else if (action == 2)
  {
    // TODO: read encoded 170x170 image from serial, plot it out
  }
  else if (action == 3)
  {
    //encoded_img = [[]];
    // TODO: pass canned encoded image to plot method
    plot_eagle();
  }
  else if (action == 4)
  {
    // do something artistic
    spiral();
  }
  delay(150);  
}

// ------------------------------------------------
void sweep()
{
  servo_pen.write(PEN_DOWN_L);
  int pos = 5;
  // trimming back the range by 5 deg on each end
  for(pos = 5; pos < 176; pos += 1)  // goes from 0 degrees to 180 degrees 
  {                                  // in steps of 1 degree 
    servo_y.write(pos);              // tell servo to go to position in variable 'pos' 
    delay(15);                       // waits 15ms for the servo to reach the position 
  } 
  for(pos = 175; pos>=5; pos-=1)     // goes from 180 degrees to 0 degrees 
  {                                
    servo_y.write(pos);              // tell servo to go to position in variable 'pos' 
    delay(15);                       // waits 15ms for the servo to reach the position 
  } 
  servo_pen.write(PEN_UP);
  delay(200);
} 


// -------------------------------------------------
void test_box()
{
    // zero out the servos
    soft_zero();
    
    servo_pen.write(PEN_DOWN_L);
    
    // step along the edges
    int s = 0;
    for (s = 5; s <= 175; s += 1)
    {
      servo_y.write(s);
      delay(15);
    }
    
    // x out
    for (s = 0; s <= 170; s += 1)
    {
      servo_x1.write(SERVO_X1_HOME + (SERVO_X1_MOVEAWAY * s));
      servo_x2.write(SERVO_X2_HOME + (SERVO_X2_MOVEAWAY * s));
      delay(15);
    }
    
    for (s = 175; s >= 5; s -= 1)
    {
      servo_y.write(s);
      delay(15);
    }
    
    // x back
    for (s = 170; s >= 0; s -= 1)
    {
      servo_x1.write(SERVO_X1_HOME + (SERVO_X1_MOVEAWAY * s));
      servo_x2.write(SERVO_X2_HOME + (SERVO_X2_MOVEAWAY * s));
      delay(15);
    }
    
    servo_pen.write(PEN_UP);
    Serial.println("box wait");
    delay(1000);
}

void spiral()
{
  // zero out the servos
    soft_zero();
    
    int xaway = 170; // relative
    int xhome = 0;
    int yaway = 170; // relative
    int yhome = 0;
    
    int spiral_step = 5; // points to reduce on each spiral
    
    boolean drag = false;
    
    // outer loop for each 
    int spin;
    for (spin = 0; spin < 10; spin += 1) 
    {
      servo_pen.write(PEN_DOWN_L);
      if (!drag) {  
        delay(PEN_WAIT);
        servo_pen.write(PEN_UP);
      }
      
      // step along the edges
      int s = 0;
      for (s = 5 + (spin * spiral_step); s <= 175 - (spin * spiral_step); s += 1)
      {
        servo_y.write(s);
        delay(15);
        if (!drag) {
          servo_pen.write(PEN_DOWN_L);
          delay(PEN_WAIT);
          servo_pen.write(PEN_UP);
          delay(15);
        }
      }
      
      // x out
      for (s = 0 + (spin * spiral_step); s <= 170 - (spin * spiral_step); s += 1)
      {
        servo_x1.write(SERVO_X1_HOME + (SERVO_X1_MOVEAWAY * s));
        servo_x2.write(SERVO_X2_HOME + (SERVO_X2_MOVEAWAY * s));
        delay(15);
        if (!drag) {
          servo_pen.write(PEN_DOWN_L);
          delay(PEN_WAIT);
          servo_pen.write(PEN_UP);
          delay(15);
        }
      }
      
      for (s = 175 - (spin * spiral_step); s >= 5 + ((spin + 1) * spiral_step); s -= 1)
      {
        servo_y.write(s);
        delay(15);
        if (!drag) {
          servo_pen.write(PEN_DOWN_L);
          delay(PEN_WAIT);
          servo_pen.write(PEN_UP);
          delay(15);
        }
      }
      
      // x back
      for (s = 170 - (spin * spiral_step); s >= 0 + ((spin + 1) * spiral_step); s -= 1)
      {
        servo_x1.write(SERVO_X1_HOME + (SERVO_X1_MOVEAWAY * s));
        servo_x2.write(SERVO_X2_HOME + (SERVO_X2_MOVEAWAY * s));
        delay(15);
      }
      
      servo_pen.write(PEN_UP);
      Serial.println("spiral wait");
      delay(PEN_WAIT);
    }
    
    // return to home nicely
    // end at home + (spin * spiral_step) in each direction, so move from there
    for (int gohomey = spin * spiral_step + 5; gohomey > 5; gohomey -= 1) 
    {
      servo_y.write(gohomey);
      delay(15);
    }
    for (int gohomex = spin * spiral_step; gohomex > 0; gohomex -= 1) 
    {
      servo_x1.write(SERVO_X1_HOME + (SERVO_X1_MOVEAWAY * gohomex));
      servo_x2.write(SERVO_X2_HOME + (SERVO_X2_MOVEAWAY * gohomex));
      delay(15);
    }
    
    delay(2000);
}

{
  Serial.println("Homing the servos");
  servo_pen.write(PEN_UP);
  servo_y.write(5);
  servo_x1.write(SERVO_X1_HOME);
  servo_x2.write(SERVO_X2_HOME);
  delay(50);
}
